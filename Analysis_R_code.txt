# Predicted fit + Wald 95% CI for a GAM via the lpmatrix (delta-method) approach.
# Returns the linear-predictor fit and SE; correct for contrasts because it uses
# the full coefficient covariance matrix rather than naive sqrt(se1^2 + se2^2).
predict_lp <- function(model, newdata, beta = coef(model), Vb = vcov(model)) {
  Xp  <- predict(model, newdata = newdata, type = "lpmatrix")
  fit <- as.vector(Xp %*% beta)
  se  <- sqrt(rowSums((Xp %*% Vb) * Xp))   # diag(Xp Vb Xp')
  data.frame(fit = fit, se = se, lower = fit - 1.96 * se, upper = fit + 1.96 * se)
}

# Mean (male) - (female) contrast at matched covariate rows, with correct SE.
sex_contrast_lp <- function(model, newdata, male = "Male", female = "Female",
                            sex_levels, beta = coef(model), Vb = vcov(model)) {
  nd_m <- transform(newdata, sex = factor(male,   levels = sex_levels))
  nd_f <- transform(newdata, sex = factor(female, levels = sex_levels))
  Xdiff <- predict(model, nd_m, type = "lpmatrix") -
    predict(model, nd_f, type = "lpmatrix")
  diff <- as.vector(Xdiff %*% beta)
  se   <- sqrt(rowSums((Xdiff %*% Vb) * Xdiff))
  data.frame(diff = diff, se = se, lower = diff - 1.96 * se, upper = diff + 1.96 * se)
}

# Equal-weight sex-averaged ("marginal") prediction with correct CI.
# Averages the linear-predictor design rows equally across sex; this equals the
# sum-to-zero effect-coding baseline and is invariant to the factor's contrast
# coding (fitted values are coding-invariant). `grid_nosex` holds one row per
# prediction point and every predictor EXCEPT sex. `sex_var` is the model's sex
# column name; `exclude` drops smooths (e.g. the site random effect) from the
# population-level prediction.
avg_over_sex <- function(model, grid_nosex, sex_levels, sex_var = "sex",
                         exclude = NULL, beta = coef(model), Vb = vcov(model)) {
  n    <- nrow(grid_nosex)
  grid <- grid_nosex[rep(seq_len(n), length(sex_levels)), , drop = FALSE]
  grid[[sex_var]] <- factor(rep(sex_levels, each = n), levels = sex_levels)
  Xp   <- predict(model, grid, type = "lpmatrix", exclude = exclude)
  grp  <- rep(seq_len(n), length(sex_levels))
  Xpm  <- rowsum(Xp, group = grp, reorder = FALSE) / length(sex_levels)  # equal avg
  fit  <- as.vector(Xpm %*% beta)
  se   <- sqrt(rowSums((Xpm %*% Vb) * Xpm))
  cbind(grid_nosex,
        data.frame(fit = fit, se = se, lower = fit - 1.96 * se, upper = fit + 1.96 * se))
}

# Tidy a GAM parametric table into term / est / se / CI / t / p with Wald CIs.
tidy_param <- function(gam_model) {
  summary(gam_model)$p.table %>%
    as.data.frame() %>%
    rownames_to_column("term") %>%
    mutate(est = Estimate, se = `Std. Error`,
           lower = est - 1.96 * se, upper = est + 1.96 * se) %>%
    dplyr::select(term, est, se, lower, upper, `t value`, `Pr(>|t|)`)
}

# Tidy a GAM smooth table into smooth / edf / ref.df / statistic / p.
tidy_smooth <- function(gam_model) {
  summary(gam_model)$s.table %>%
    as.data.frame() %>%
    rownames_to_column("smooth")
}

cross_sens <- purrr::map_dfr(sensitivity_preds, function(pred) {
  message("Cross-sectional sensitivity model: ", pred)
  fmla <- as.formula(paste0(
    outcome, " ~ ",
    "s(age, by = sex_unordered, k = 20, bs = 'cr') + ",
    "s(age, k = 20, bs = 'cr') + ",
    "sex_unordered * ", pred, " + ",
    "s(site, bs = 're')"
  ))
  mod <- bam(fmla, data = ng1, method = "fREML", na.action = na.exclude)
  
  summary(mod)$p.table %>%
    as.data.frame() %>%
    rownames_to_column("term") %>%
    # Keep the sex x predictor interaction term(s) for this model
    filter(str_detect(term, "^sex_unorderedMale:")) %>%
    mutate(predictor = pred,
           est = Estimate, se = `Std. Error`,
           lower = est - 1.96 * se, upper = est + 1.96 * se) %>%
    dplyr::select(predictor, term, est, se, lower, upper,
                  `t value`, p = `Pr(>|t|)`)
})

sex_levels <- levels(df_long$sex)

# Random-effects structure used throughout: random intercept + slope of time
# per participant, plus a random intercept for assessment centre.
random_effects <- ~ (1 + time_c | ID_subject) + (1 | UK_Biobank_assessment_centre)
k_basis <- 20

# Shared plotting grids
age_mean   <- mean(df_long$age_at_scan, na.rm = TRUE)
time_seq   <- seq(-3.5, 3.5, by = 0.1)   # within observed within-person range

# -----------------------------------------------------------------------------
# 5b. Overall longitudinal model (averaged across sexes)  -> Figure 3A / 3B
#     Single shared s(mean_age_c) + sum-to-zero (effect) coded sex; no sex
#     interactions. Because there is no time:sex term, the within-person time
#     effect (time_c) and its age moderation (time_c:mean_age_c), and the
#     s(mean_age_c) edf/p, are INVARIANT to the sex coding; effect coding only
#     relocates the intercept to the sex-averaged grand mean.
# -----------------------------------------------------------------------------
gamm_main <- gamm4(
  QSM_adj ~ s(mean_age_c, k = k_basis, bs = "cr") +
    sex_sum + time_c + time_c:mean_age_c,
  random  = random_effects,
  data    = df_long, REML = TRUE,
  control = lmerControl(check.nobs.vs.nRE = "ignore")
)
summary(gamm_main$gam)

# Overall longitudinal statistics (time_c = within-person change/yr;

# -----------------------------------------------------------------------------
# 5d. Longitudinal sensitivity analyses: time x sex x predictor (3-way)
#     The 3-way contrast is the difference, between predictor levels, in the
#     sex difference of the within-person time slope. Computed via a numeric
#     derivative of the lpmatrix in time so it is exact under the fitted model.
# -----------------------------------------------------------------------------
# Factor-code categorical moderators
for (v in c("family_history_PD", "new_smoking",
            "Alcohol_intake_frequency_Instance_2_edited", "Higher_education_binary",
            "manifest_CVD_instance2_with_self_report",
            "manifest_diabetes_instance2_with_self_report")) {
  df_long[[v]] <- factor(df_long[[v]])
}

delta <- 1e-3  # step for the numeric time-derivative

# Time-slope linear-predictor row for a given sex and predictor value
slope_row <- function(model, sex_level, pred, pred_value) {
  nd_p <- data.frame(mean_age_c = 0, time_c =  delta,
                     sex = factor(sex_level, levels = sex_levels))
  nd_m <- data.frame(mean_age_c = 0, time_c = -delta,
                     sex = factor(sex_level, levels = sex_levels))
  if (is.factor(df_long[[pred]])) {
    nd_p[[pred]] <- factor(pred_value, levels = levels(df_long[[pred]]))
    nd_m[[pred]] <- factor(pred_value, levels = levels(df_long[[pred]]))
  } else {
    nd_p[[pred]] <- pred_value; nd_m[[pred]] <- pred_value
  }
  (predict(model, nd_p, type = "lpmatrix") -
      predict(model, nd_m, type = "lpmatrix")) / (2 * delta)
}

long_sens <- purrr::map_dfr(sensitivity_preds, function(pred) {
  message("Longitudinal sensitivity model: ", pred)
  fmla <- as.formula(paste0(
    "QSM_adj ~ s(mean_age_c, by = sex, k = ", k_basis, ", bs = 'cr') + ",
    "sex + time_c + time_c:sex + ",
    pred, " + sex:", pred, " + time_c:", pred, " + time_c:sex:", pred
  ))
  mod <- tryCatch(
    gamm4(fmla, random = random_effects, data = df_long, REML = TRUE,
          control = lmerControl(check.nobs.vs.nRE = "ignore"),
          na.action = na.exclude),
    error = function(e) { warning("Failed for ", pred, ": ", conditionMessage(e)); NULL }
  )
  if (is.null(mod)) return(NULL)
  
  g <- mod$gam; b <- coef(g); V <- vcov(g)
  
  # Determine predictor levels/values to contrast
  if (is.factor(df_long[[pred]])) {
    lv  <- levels(df_long[[pred]]); ref <- lv[1]; alts <- lv[-1]
  } else {
    ref <- 0; alts <- 1
  }
  
  purrr::map_dfr(alts, function(lvl) {
    L <- (slope_row(g, "Male",   pred, lvl) - slope_row(g, "Female", pred, lvl)) -
      (slope_row(g, "Male",   pred, ref) - slope_row(g, "Female", pred, ref))
    est <- as.numeric(L %*% b)
    se  <- sqrt(rowSums((L %*% V) * L))
    t   <- est / se
    tibble(predictor = pred,
           level = as.character(lvl),
           est = est, se = se,
           lower = est - 1.96 * se, upper = est + 1.96 * se,
           t = t, p = 2 * pnorm(-abs(t)))
  })
})

# FDR across the family of 3-way contrasts
long_sens <- long_sens %>%
  mutate(p_adj_fdr = p.adjust(p, method = "fdr"),
         label = pred_labels[predictor])

write.csv(long_sens,
          file.path(long_dir, "longitudinal_sensitivity_threeway_contrasts.csv"),
          row.names = FALSE)
